-- Create Database

CREATE DATABASE B_SKIN_SERVER;


-- Create Providers Table

CREATE TABLE BS_PROVIDERS (
ID BIGINT IDENTITY(1,1) PRIMARY KEY,
NAME VARCHAR(100) NOT NULL,
DESCRIPTION VARCHAR(500) NOT NULL,
DOCUMENT VARCHAR(14) NOT NULL,
CREATED_ON DATE NOT NULL,
IS_ACTIVE BIT DEFAULT 1
);


-- Create T-Shirt Table

CREATE TABLE BS_TSHIRTS (
  ID BIGINT IDENTITY(1,1) PRIMARY KEY,
  NAME VARCHAR(100) NOT NULL,
  DESCRIPTION VARCHAR(500) NOT NULL,
  PRICE DECIMAL(10,2) NOT NULL,
  QUANTITY_IN_STOCK INT DEFAULT 0,
  CREATED_ON DATE NOT NULL,
  IS_ACTIVE BIT DEFAULT 1,
  PROVIDER_ID BIGINT NOT NULL,
  CONSTRAINT FK_PROVIDER_ID FOREIGN KEY (PROVIDER_ID) REFERENCES BS_PROVIDERS (ID)
);


-- Create Provider Type Table

CREATE TABLE BS_PROVIDER_TYPE (
  `ID` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
  `TYPE` VARCHAR(50) NOT NULL,
  PRIMARY KEY (`ID`)
);
